    }
    allow = {
        event_target:caozuo_fleet_ing = {
            not = {
                has_fleet_flag = fleet_is_concontroller
            }
            not = {
                is_owned_by = from
            }
        }
        from = {
            OR = {
                has_country_flag = axis_fleet_1
                has_country_flag = axis_fleet_2
                has_country_flag = axis_fleet_3
                has_country_flag = axis_fleet_4
                has_country_flag = axis_fleet_5
                has_country_flag = axis_fleet_6
            }
        }
    }
    effect = {
        event_target:caozuo_fleet_ing = {
            give_fleet = {
                controller = from
                days = 10
            }
            set_timed_fleet_flag = {
                flag = fleet_is_concontroller
                days = 10
            }
        }
    }
}

axis_fleet_effect_3 = {
    #显示条件
    potential = {
        show_axis_fleet = yes
    }
    allow = {
        event_target:caozuo_fleet_ing = {
            is_owned_by = from
        }

        from = {
            OR = {
                has_country_flag = axis_fleet_1
                has_country_flag = axis_fleet_2
                has_country_flag = axis_fleet_3
                has_country_flag = axis_fleet_4
                has_country_flag = axis_fleet_5
                has_country_flag = axis_fleet_6
            }
        }
    }
    effect = {
        from = {
            #没有出租就清理位置
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_1
                }
                event_target:axis_fleet_1 = {
                    remove_fleet_flag = axis_fleet_1
                }
                clear_global_event_target = axis_fleet_1
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_1
            }
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_2
                }
                event_target:axis_fleet_2 = {
                    remove_fleet_flag = axis_fleet_2
                }
                clear_global_event_target = axis_fleet_2
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_2
            }
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_3
                }
                event_target:axis_fleet_3 = {
                    remove_fleet_flag = axis_fleet_3
                }
                clear_global_event_target = axis_fleet_3
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_3
            }
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_4
                }
                event_target:axis_fleet_4 = {
                    remove_fleet_flag = axis_fleet_4
                }
                clear_global_event_target = axis_fleet_4
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_4
            }
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_5
                }
                event_target:axis_fleet_5 = {
                    remove_fleet_flag = axis_fleet_5
                }
                clear_global_event_target = axis_fleet_5
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_5
            }
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        not = {
                            has_fleet_flag = fleet_is_concontroller
                        }
                    }
                    has_country_flag = axis_fleet_6
                }
                event_target:axis_fleet_6 = {
                    remove_fleet_flag = axis_fleet_6
                }
                clear_global_event_target = axis_fleet_6
                clear_global_event_target = caozuo_fleet_ing
                remove_global_flag = has_axis_fleet_6
            }
            #在出租就回收
            if = {
                limit = {
                    event_target:caozuo_fleet_ing = {
                        has_fleet_flag = fleet_is_concontroller
                    }
                }
                event_target:caozuo_fleet_ing = {
                    end_fleet_contract = {
                        initiator = from
                        reason = broken
                    }
                    remove_fleet_flag = fleet_is_concontroller
                }
            }
            # country_event = { id = ammunition.103 }
        }
    }
}
