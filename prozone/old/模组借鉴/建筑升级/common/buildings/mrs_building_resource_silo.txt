# 定义
building_resource_silo = {
    # 基础建造时间（秒），180秒完成建造
    base_buildtime = 180
    
    # 建筑分类：资源类建筑
    category = resource
    
    # 潜在可用条件（排除特定殖民地类型）
    potential = {
        NOR = {  # 排除以下三种殖民地类型
            has_modifier = resort_colony     # 度假殖民地（不可建造）
            has_modifier = slave_colony      # 奴隶殖民地（不可建造）
            has_modifier = crucible_colony   # 熔炉殖民地（不可建造）
        }
    }
    
    # 建造允许条件（核心逻辑）
    allow = {
        exists = owner  # 必须存在星球所有者
        # 隐藏触发条件（AI控制或满足特殊条件时允许建造）
        hidden_trigger = {
            OR = {
                # 玩家控制时直接允许
                owner = { is_ai = no }
                # 满足复合条件时允许（主要针对AI建造限制）
                AND = {
                    # 未建造同类型建筑（避免重复建造）
                    NOT = { has_building = building_resource_silo }
                    OR = {  # 满足以下任一特殊条件：
                        # 机器帝国且拥有"模块化仓库"传统
                        owner = {
                            is_machine_empire = yes
                            has_active_tradition = tr_domination_modular_depot
                        }
                        # 激活银河共同体决议"平衡之道"
                        is_active_resolution = "resolution_greatergood_balance_in_the_middle"
                        # 激活银河共同体决议"普遍繁荣法令"
                        is_active_resolution = "resolution_greatergood_universal_prosperity_mandate"
                        # 拥有"巨型都市计划"飞升权限但未研发巨型工程
                        owner = {
                            has_ascension_perk = ap_arcology_project
                            NOT = { has_technology = tech_mega_engineering }
                        }
                    }
                }
            }
        }
    }
    
    # 行星级触发修饰符（常规帝国适用）
    triggered_planet_modifier = {
        potential = {
            exists = owner
            # 仅对常规帝国类型生效（非格式塔/机器帝国）
            owner = { is_regular_empire = yes }
        }
        modifier = {
            job_clerk_add = 1  # 增加1个文员岗位（提供基础资源产出）
        }
    }
    
    # 行星级触发修饰符（格式塔意识适用）
    triggered_planet_modifier = {
        potential = {
            exists = owner
            # 仅对格式塔意识类型生效（蜂群/机飞）
            owner = { is_gestalt = yes }
        }
        modifier = {
            job_maintenance_drone_add = 1  # 增加1个维护无人机岗位（保障建筑运转）
        }
    }
    
    # 资源消耗与维护成本
    resources = {
        category = planet_buildings  # 属于行星建筑类资源消耗
        cost = {
            minerals = 200  # 建造消耗200矿物资源
        }
        upkeep = {
            energy = 1  # 每周期（月）消耗1能源维持运转
        }
    }
    
    # 国家级修饰符（永久生效）
    country_modifier = {
        country_resource_max_add = 5000  # 国家资源存储上限永久增加5000单位
    }
    
    # 条件触发国家修饰符（科技研发后生效）
    triggered_country_modifier = {
        potential = {
            exists = owner
            # 需研发"建造模板"科技（高级建筑规划科技）
            owner = { has_technology = tech_construction_templates }
        }
        country_resource_max_add = 5000  # 额外增加5000资源上限（总计10000）
    }
    
    # 科技前置条件（必须先研发基础工业科技）
    prerequisites = {
        "tech_basic_industry"  # 基础工业科技（解锁基础建造能力）
    }
    
    # 建筑升级路径（可升级至中级资源仓库）
    upgrades = {
        mrs_building_resource_silo  # 指向中级资源仓库的升级选项
    }
    
    # 触发式建筑描述（格式塔意识专用）
    triggered_desc = {
        trigger = {
            exists = owner
            owner = { is_gestalt = yes }  # 格式塔意识触发
        }
        text = job_maintenance_drone_effect_desc  # 显示维护无人机岗位效果描述
    }
    
    # 触发式建筑描述（常规帝国专用）
    triggered_desc = {
        trigger = {
            exists = owner
            owner = { is_gestalt = no }  # 非格式塔意识触发
        }
        text = job_clerk_effect_desc  # 显示文员岗位效果描述
    }
}
